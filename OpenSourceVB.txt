Imports Excel = Microsoft.Office.Interop.Excel
Public Class Form1
    Dim len(32767) As Integer
    Dim ind(2047) As Integer
    Dim names(32767) As String
    Dim nick(32767) As String
    Dim i, j, cs, rs As Integer
    Private Sub Button1_Click(ByVal sender As System.Object, ByVal e As System.EventArgs) Handles Button1.Click
        Dim xlApp As Excel.Application
        Dim xlWorkBook As Excel.Workbook
        Dim xlWorkSheet As Excel.Worksheet
        OpenFileDialog.InitialDirectory = My.Computer.FileSystem.SpecialDirectories.MyDocuments
        OpenFileDialog.Filter = "XLS Files (*.xls)|*xls"
        OpenFileDialog.FileName = ""
        If (OpenFileDialog.ShowDialog(Me) = System.Windows.Forms.DialogResult.OK) Then
            DGV.SelectAll()
            DGV.ClearSelection()
            DGV.Rows.Clear()
            Dim FileName As String = OpenFileDialog.FileName
            xlApp = New Excel.Application
            xlWorkBook = xlApp.Workbooks.Open(FileName)
            xlWorkSheet = xlWorkBook.Worksheets("Лист1")
            'Определение количества строк в таблице
            For i = 1 To 256
                If xlWorkSheet.Cells(i, 1).value Is Nothing Then
                    rs = i - 1
                    i = 257
                End If
            Next
            ' экспорт таблицы в DataGridView
            For i = 1 To rs
                DGV.Rows.Add(xlWorkSheet.Cells(i, 1).value, xlWorkSheet.Cells(i, 2).value, xlWorkSheet.Cells(i, 3).value, xlWorkSheet.Cells(i, 4).value)
            Next
            ' закрыть таблицу Exel
            ' MsgBox("Диапазон таблицы:" + cs.ToString + "x" + rs.ToString)
            ' xlWorkSheet.Cells(2, 2) = "http://vb.net-informations.com"
            xlWorkBook.Close()
            xlApp.Quit()
            releaseObject(xlApp)
            releaseObject(xlWorkBook)
            releaseObject(xlWorkSheet)
            ' сортировка DataGridView по шифру профиля
            DGV.Sort(DGV.Columns(0), System.ComponentModel.ListSortDirection.Ascending)
            DGV.Refresh()
            rs = DGV.RowCount
            Button2.Enabled = (rs - 1 > 0)
        End If
    End Sub

    Private Sub Button2_Click(ByVal sender As System.Object, ByVal e As System.EventArgs) Handles Button2.Click
        DGV.Sort(DGV.Columns(0), System.ComponentModel.ListSortDirection.Ascending)
        DGV.Refresh()
        TextBox5.Clear()
        ' формированик массивов: '
        ' len - массив деталей;
        ' ind - диапазоны элементов по шифрам профилей;
        ' names - шифры профилей.
        Dim nc, k, l, total As Integer
        Dim s As String
        Button2.Enabled = False
        i = 0
        j = 0
        k = 0
        ind(0) = 0
        total = 0
        rs = DGV.RowCount
        ProgressBar1.Visible = True
        Do While i < rs - 1
            s = CType(DGV.Rows(i).Cells(0).Value, String)
            names(j) = s
            Do While (s = CType(DGV.Rows(i).Cells(0).Value, String))
                nc = nc + Val(CType(DGV.Rows(i).Cells(2).Value, String))
                For l = 1 To Val(CType(DGV.Rows(i).Cells(2).Value, String))
                    len(k) = Val(CType(DGV.Rows(i).Cells(1).Value, String))
                    nick(k) = CType(DGV.Rows(i).Cells(3).Value, String)
                    If len(k) + Val(TextBox2.Text) >= Val(TextBox1.Text) Then
                        MsgBox("Длина больше заданной " + s + " L=" + len(k).ToString + "+" + TextBox2.Text + ">" + TextBox1.Text)
                        Return
                    End If
                    'TextBox5.AppendText(vbCrLf + s + vbTab + " | длина " + CType(DGV.Rows(i).Cells(1).Value, String) + vbTab + " | индекс " + k.ToString)
                    k += 1
                Next
                If i < rs - 1 Then i += 1
            Loop
            'TextBox5.AppendText(vbCrLf + "Профиль " + s + vbTab + " | Количество деталей " + nc.ToString)
            total = total + nc
            nc = 0
            j += 1
            ind(j) = k
        Loop
        'TextBox5.AppendText(vbCrLf + "Позиций " + j.ToString)
        'For i = 0 To j
        'TextBox5.AppendText(vbCrLf + "индекс " + i.ToString + " " + ind(i).ToString)
        'Next
        Dim SDelta As Integer
        Dim SIndex As Integer
        Dim SDets(2047) As Integer

        Dim NDetsmin As Integer
        Dim NInDetsMin(2047) As Integer
        Dim DeltaMin(2047) As Integer
        Dim CDetsMin(2047, 2047)
        Dim pos As Integer = j
        Dim cutlength As Integer
        Dim lenmax As Integer
        Dim Delta(2047) As Integer
        Dim CLens(total) As Integer
        Dim CDets(2047, 2047) As Integer
        Dim NInDets(2047) As Integer
        Dim NLens, NTmp, ITmp, NDets As Integer
        Dim result, resultmin As Double
        Dim Itt, ittmax As Integer
        cutlength = Val(TextBox2.Text)
        lenmax = Val(TextBox1.Text)
        ittmax = Val(TextBox3.Text)
        If ittmax < 1 Then
            ittmax = 100
        Else
            ittmax = ittmax * 100
        End If
        For l = 0 To pos - 1
            resultmin = 100
            NLens = 0
            For i = ind(l) To ind(l + 1) - 1
                NLens += 1
            Next
            For Itt = 0 To ittmax * NLens
                NDets = 0
                NLens = 0
                For i = ind(l) To ind(l + 1) - 1
                    CLens(i - ind(l)) = len(i)
                    NLens += 1
                Next
                NDets = 0
                ITmp = Int(Rnd() * NLens)
                Delta(0) = lenmax
                NTmp = NLens
                Do While NTmp > 0
                    ITmp = Int(Rnd() * NLens)
                    Do While CLens(ITmp) <> 0
                        For i = 0 To NDets
                            If Delta(i) >= CLens(ITmp) + cutlength Then
                                Delta(i) = Delta(i) - CLens(ITmp) - cutlength
                                CDets(i, NInDets(i)) = ITmp
                                NInDets(i) += 1
                                CLens(ITmp) = 0
                                NTmp -= 1
                                i = NDets + 1
                            End If
                        Next
                        If CLens(ITmp) > 0 Then
                            NDets += 1
                            Delta(NDets) = lenmax
                            NInDets(NDets) = 0
                        End If
                    Loop
                Loop
                result = 0
                For i = 0 To NDets
                    result = result + Delta(i)
                Next
                result = (result) * 100 / (lenmax * (NDets + 1))
                If result < resultmin Then
                    resultmin = result
                    NDetsmin = NDets
                    For i = 0 To NDets
                        NInDetsMin(i) = NInDets(i)
                        DeltaMin(i) = Delta(i)
                        For j = 0 To NInDets(i) - 1
                            CDetsMin(i, j) = CDets(i, j)
                        Next
                    Next
                End If
                For i = 0 To NDets
                    NInDets(i) = 0
                Next
                ProgressBar1.Value = Itt * 100 / (ittmax * NLens)
            Next
            TextBox5.AppendText(vbCrLf + " Результат:" + names(l) + " отходы:" + resultmin.ToString("f3") + ", кол-во заготовок:" + (NDetsmin + 1).ToString + ", деталей:" + NLens.ToString)
            ' сортировка длин деталей в заготовке
            Dim t2 As Integer
            For i = 0 To NDetsmin
                For k = 0 To NInDetsMin(i) - 1
                    For j = k To NInDetsMin(i) - 1
                        If len(CDetsMin(i, j) + ind(l)) > len(CDetsMin(i, k) + ind(l)) Then
                            t2 = CDetsMin(i, j)
                            CDetsMin(i, j) = CDetsMin(i, k)
                            CDetsMin(i, k) = t2
                        End If
                    Next
                Next
            Next
            ' сортировка заготовок по длинам первого элемента
            For i = 0 To NDetsmin
                For j = i To NDetsmin
                    If (len(CDetsMin(j, 0) + ind(l)) * 2047 - NInDetsMin(j)) > (len(CDetsMin(i, 0) + ind(l)) * 2047 - NInDetsMin(i)) Then
                        ' копирование массивов
                        For k = 0 To NInDetsMin(i) - 1
                            SDets(k) = CDetsMin(i, k)
                        Next
                        SIndex = NInDetsMin(i)
                        SDelta = DeltaMin(i)
                        For k = 0 To NInDetsMin(j) - 1
                            CDetsMin(i, k) = CDetsMin(j, k)
                        Next
                        NInDetsMin(i) = NInDetsMin(j)
                        DeltaMin(i) = DeltaMin(j)
                        For k = 0 To SIndex
                            CDetsMin(j, k) = SDets(k)
                        Next
                        NInDetsMin(j) = SIndex
                        DeltaMin(j) = SDelta
                    End If
                Next
            Next
            For i = 0 To NDetsmin
                TextBox5.AppendText(vbCrLf + " деталь:" + (i + 1).ToString + vbTab + ", остаток:" + DeltaMin(i).ToString)
                For j = 0 To NInDetsMin(i) - 1
                    TextBox5.AppendText(vbTab + "   " + Strings.Left(nick(CDetsMin(i, j) + ind(l)), 4))
                Next
                TextBox5.AppendText(vbCrLf + "        " + vbTab + "          ")
                For j = 0 To NInDetsMin(i) - 1
                    TextBox5.AppendText(vbTab + " L=" + len(CDetsMin(i, j) + ind(l)).ToString)
                Next
            Next
            TextBox5.AppendText(vbCrLf)
        Next
        ProgressBar1.Visible = False
        TextBox5.AppendText(vbCrLf + " Расчет завершен")
        Button2.Enabled = rs - 1 > 0
    End Sub

    Private Sub releaseObject(ByVal obj As Object)
        Try
            System.Runtime.InteropServices.Marshal.ReleaseComObject(obj)
            obj = Nothing
        Catch ex As Exception
            obj = Nothing
        Finally
            GC.Collect()
        End Try
    End Sub

        Button2.Enabled = rs - 1 > 0
    End Sub
End Class
